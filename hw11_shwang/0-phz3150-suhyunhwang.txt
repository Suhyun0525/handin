def ellipse(x, y, a, b):
    """Function that plots an ellipse
    
    Input: x(x-coordinate), y(y-coordinate), a(width), b(height)
    Output: plot   """
    
    x,y = np.meshgrid(x,y)
    
    z = (( x**2 / a**2 ) + ( y**2 / b**2 ) - 1 ) 
    
    fig, ax = plt.subplots()
    
    ax.contour(x,y,z,[0])
   # ax.set_aspect(1)
    plt.xlim(-15,15)
    plt.ylim(-25,25)

    for i in range(len(input_array)):
        m_index = i
        for j in range(i, len(input_array)):
            if input_array[j] < input_array[m_index]:

class holiday_frenzy:
    def __init__(self,a,b,c, number_of_gifts, wrap_paper_available):
        self.dimension1 = a
        self.dimension2 = b
        self.dimension3 = c
        self.gifts = number_of_gifts
        self.paper = wrap_paper_available
     
    def surface_area(self):
        area = self.dimension1 * self.dimension2 
        return area
       
    def volume(self):
        volume = self.surface_area() * self.dimension3
        return volume
    
    def enough_paper(self):
        if self.volume() <= self.paper:
            return True
        else:
            return False
    
    def fits(self):
        if self.volume() >= self.gifts * 25:
            return True
        else:
            return False
            
    def print_gift(self):
        print("Your box surface area is", self.surface_area(), "inches^2 and its volume is", self.volume(), "inches^3.")
        
        if self.enough_paper() == True:

        else:
            print("You don't have enough paper to wrap your box.")
            
        if self.fits() == True:
            print("Your gifts fit in the box.")

    """Function that calculates the pressures for the piston volumes in V, using the adiabatic law and 
    the temperatures using the ideal gas law.
    Input: an array of N volumes, initial (P; V; T) values
    Output: Pressures of the piston & the temperatures """
    
    for i in V:

        T = (T0 * P * i) / (P * V0)

    """Function that calculates the wind chill factor.
    Input: the actual temperature T and the wind speed V
    Output: wind chill temperature WC """
    
    WC = 35.74 + 0.6215 * T - 35.75 * V**0.16 + 0.4275 * T * V**0.16
